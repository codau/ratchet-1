Basic Example #3

            $(document).ready(function() {
                $('#sample').ratchet(function() {
                    var lorem = "Lorem ipsum dolor sit amet, consectetuer adipiscing elit.";
                    this.get("/", function(el) {
                        for (var i = 0; i < 10; i++) {
                            $("
").html(lorem).appendTo(el);
                        }
                        // swap our content in
                        el.swap();
                    });
                }).run("/");
            });
