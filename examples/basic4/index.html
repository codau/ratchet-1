Basic Example #4

$(document).ready(function() {
    $('#sample').ratchet(
        function() {
            this.get("/", function(el) {
                el.html("Click a page")
                .append("<br/>")
                .append("<a href='#/page1'>Page1</a>")
                .append("<br/>")
                .append("<a href='#/page2'>Page2</a>");

                // swap our content in
                el.swap();
            });

            this.get("/page1", function(el) {
                el.html("You clicked page 1").swap();
            });

            this.get("/page2", function(el) {
                el.html("You clicked page 2").swap();
            }
        );
    }).run("/");
});
