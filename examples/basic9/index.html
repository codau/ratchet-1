Basic Example #9

$(document).ready(function() {
    $('#sample').ratchet(function() {
        this.get("/", function(el) {
            el.fetch("data.json", function(el, json) {
                el.transform("template", json, function(el) {
                    el.swap();
                });
            });
        });
    }).run("/");
});
